import requests
import json

def emotion_detector(text_to_analyse):
    """
    Detects emotions from the given text by making a POST request to the Watson NLP API.
    Formats the response to return individual emotion scores and the dominant emotion.
    """
    # URL and Headers required by the Watson NLP Emotion Predict service
    url = 'https://sn-watson-emotion.labs.skills.network/v1/watson.runtime.nlp.v1/NlpService/EmotionPredict'
    headers = {"grpc-metadata-mm-model-id": "emotion_aggregated-workflow_lang_en_stock"}
    
    # Input payload
    input_json = { "raw_document": { "text": text_to_analyse } }
    
    # Send POST request
    response = requests.post(url, json=input_json, headers=headers)
    
    # Parse the text from the response into a dictionary
    formatted_response = json.loads(response.text)
    
    # If the response is successful (status code 200)
    if response.status_code == 200:
        # Extract the specific emotion scores from the Watson API JSON structure
        emotions = formatted_response['emotionPredictions'][0]['emotion']
        
        anger_score = emotions['anger']
        disgust_score = emotions['disgust']
        fear_score = emotions['fear']
        joy_score = emotions['joy']
        sadness_score = emotions['sadness']
        
        # Create a dictionary of just the scores to find the maximum value
        emotion_scores = {
            'anger': anger_score,
            'disgust': disgust_score,
            'fear': fear_score,
            'joy': joy_score,
            'sadness': sadness_score
        }
        
        # Determine the dominant emotion by finding the key with the highest value
        dominant_emotion = max(emotion_scores, key=emotion_scores.get)
        
        # Return the final formatted dictionary
        return {
            'anger': anger_score,
            'disgust': disgust_score,
            'fear': fear_score,
            'joy': joy_score,
            'sadness': sadness_score,
            'dominant_emotion': dominant_emotion
        }
        
    # Handle cases where the input is invalid (like empty text)
    elif response.status_code == 400:
        return {
            'anger': None,
            'disgust': None,
            'fear': None,
            'joy': None,
            'sadness': None,
            'dominant_emotion': None
        }
